  <script>
    async function loadQueue() {
      const list = document.getElementById('queueList');
      list.innerHTML = 'กำลังโหลดข้อมูล...';

      try {
        const res = await fetch(API_URL + '?action=getQueue');
        const data = await res.json();

        if (!data.queue || data.queue.length === 0) {
          list.innerHTML = '<div class="empty">ยังไม่มีคิว</div>';
          return;
        }

        list.innerHTML = data.queue.map((q, index) => `
          <div class="queue-item">
            <div class="name">คิวที่ ${index + 1}: ${q.customer || '-'}</div>
            <div class="meta">
              ประเภทงาน: ${q.type || '-'}<br>
              ช่องทาง: ${q.channel || '-'}<br>
              รายละเอียด: ${q.detail || '-'}<br>
              กำหนดส่ง: ${q.deadline || '-'}<br>
              ด่วน: ${q.urgent || '-'}
            </div>
            <div class="row">
              <button class="btn-del" onclick="deleteQueue(${index + 2})">ลบคิวนี้</button>
            </div>
          </div>
        `).join('');
      } catch (err) {
        list.innerHTML = '<div class="empty">โหลดข้อมูลไม่สำเร็จ กรุณาตรวจสอบ API หรือสิทธิ์การเข้าถึง</div>';
      }
    }

    async function addQueue() {
      const customer = document.getElementById('customer').value.trim();
      const type = document.getElementById('type').value.trim();
      const channel = document.getElementById('channel').value.trim();
      const detail = document.getElementById('detail').value.trim();
      const deadline = document.getElementById('deadline').value.trim();
      const urgent = document.getElementById('urgent').value.trim();

      if (!customer) {
        alert('กรุณากรอกชื่อลูกค้า');
        return;
      }

      const url = new URL(API_URL);
      url.searchParams.set('action', 'addQueue');
      url.searchParams.set('customer', customer);
      url.searchParams.set('type', type);
      url.searchParams.set('channel', channel);
      url.searchParams.set('detail', detail);
      url.searchParams.set('deadline', deadline);
      url.searchParams.set('urgent', urgent);

      try {
        const res = await fetch(url.toString());
        const data = await res.json();

        if (data.success) {
          alert('เพิ่มคิวเรียบร้อย');
          document.getElementById('customer').value = '';
          document.getElementById('type').value = '';
          document.getElementById('channel').value = '';
          document.getElementById('detail').value = '';
          document.getElementById('deadline').value = '';
          document.getElementById('urgent').value = '';
          loadQueue();
        } else {
          alert('เพิ่มคิวไม่สำเร็จ');
        }
      } catch (err) {
        alert('เชื่อมต่อ API ไม่สำเร็จ');
      }
    }

    async function deleteQueue(row) {
      if (!confirm('ต้องการลบคิวนี้ใช่ไหม?')) return;

      const url = new URL(API_URL);
      url.searchParams.set('action', 'deleteQueue');
      url.searchParams.set('row', row);

      try {
        const res = await fetch(url.toString());
        const data = await res.json();

        if (data.success) {
          alert('ลบคิวเรียบร้อย');
          loadQueue();
        } else {
          alert('ลบคิวไม่สำเร็จ');
        }
      } catch (err) {
        alert('เชื่อมต่อ API ไม่สำเร็จ');
      }
    }

    loadQueue();
  </script>
